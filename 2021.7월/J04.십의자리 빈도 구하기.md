# 십의 자리 빈도 구하기

```java
package com.ssafy.algo;
import java.util.Scanner;
//1번 문제
public class DigitTest1 {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int[] Arr = new int[10];
		while ( true ) {
			int input = sc.nextInt();
			if (input == 0)
				break;
			else {
				int temp = input / 10;
				Arr[temp]++;
			}
		}
		for (int i=0; i<10; i++) {
			if (Arr[i] == 0 )
					continue;
			else 
				System.out.println(i + " : " + Arr[i] + "개");
		}
	}
}
```
