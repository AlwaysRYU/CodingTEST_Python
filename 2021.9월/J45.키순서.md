### 풀이
* 이건 시간초과가 안나지만, 밑의 코드는 시간초과가 난다.
* 둘의 차이는 check를 어디에서 하냐에 따라 다르다.
* check 위치가 시간에 크게 영향을 주는 것일까? 내일 교수님께 물어봐야겠다.
```java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class Solution {
	static int N;
	static int smallpeople, bigpeople;
	static boolean[][] arr;
	
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		int Test = Integer.parseInt(br.readLine());
		for (int TT = 1; TT <= Test; TT++) {
			
			// 사람 수 
			N = Integer.parseInt(br.readLine());
			arr = new boolean[N+1][N+1];
			
			// 키 비교횟수
			int M = Integer.parseInt(br.readLine());
			for (int i = 0; i < M; i++) {
				StringTokenizer st = new StringTokenizer(br.readLine());
				int a = Integer.parseInt(st.nextToken());
				int b = Integer.parseInt(st.nextToken());
				arr[a][b] = true;	
			}
			// 입력 완료
			
			// 연산 시작
			int answer = 0;
			for (int i = 1; i <= N; i++) {
				// 비교해서 
				smallpeople = bigpeople = 0;
				find_small(i);
				find_big(i);
//				System.out.println(i + "번의 작은사람 : " + S + " 큰사람 : " + B);
				if ( smallpeople+bigpeople == N-1) answer += 1; 
				// 알수있는 사람 수가  N-1 이면  answer += 1;
			}
			
			System.out.println("#" + TT + " " + answer);
		}

		
		
		
	}

	static void find_small(int x) {
		boolean[] check = new boolean[N+1];
		
		Queue<Integer> Q = new LinkedList<>();
		Q.offer(x);
		check[x] = true;
		
		while( !Q.isEmpty() ) {
			int now = Q.poll();
			for (int i = 1; i <= N; i++) {
				if (arr[i][now] && !check[i]) {
					smallpeople += 1;
					check[i] = true;
					Q.offer(i);
				}
			}
		}
		
	}
	
	static void find_big(int x) {
		boolean[] check = new boolean[N+1];
		
		Queue<Integer> Q = new LinkedList<>();
		Q.offer(x);
		check[x] = true;
		while( !Q.isEmpty() ) {
			int now = Q.poll();
			for (int i = 1; i <= N; i++) {
				if (arr[now][i] && !check[i]) {
					bigpeople += 1;
					check[i] = true;
					Q.offer(i);
				}
			}
		}
	}
		
}
```

### 다른방법
* DFS를 사용 
* 행렬을 두개만들어서 메소드는 1개로 구현
* DP를 사용 계속 인접행렬을 업데이트하기
```java
package R12_문제풀이;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;
import java.util.StringTokenizer;

public class SOL01_키순서_DP {
	static int ans, N, M, adj[][];
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		int TC = Integer.parseInt(br.readLine());
		for (int test = 1; test <= TC; test++) {
			
			N = Integer.parseInt(br.readLine());
			M = Integer.parseInt(br.readLine());
			adj = new int[N+1][N+1];
			
			StringTokenizer st = null;
			for (int m = 0; m < M; m++) {
				 st = new StringTokenizer(br.readLine());
				 int from = Integer.parseInt(st.nextToken());
				 int to = Integer.parseInt(st.nextToken());
				 adj[from][to] = 1;
			}
			
			for (int i = 1; i <= N; i++) {
				adj[i][0] = -1;
				// -1이면 탐색을 하지 않음
			}
			
			ans = 0;
			for (int i = 1; i <= N; i++) {
				if(adj[i][0] != -1) DFS(i);
			}
			System.out.println(ans);
		}
		
	}
	
	private static void DFS(int cur) {
		for (int i = 1; i <= N; i++) {
			if( adj[cur][i] == 1) {
				// 자신보다 큰 학생을 만났으면
				if ( adj[i][0] != -1 ) {
					// 탐색 전
					DFS(i);
				}
				// 자신보다 큰 학생 탐색을 완료했음.
				// 메모가 되어있으면 탐색 안하고 바로 내려온다.
				if ( adj[i][0] > 0 ) {
					// i 보다 큰 학생이 존재한다.
					// i의 인접행렬 상태를 cur에 반영한다.
					for (int j = 1; j <= N; j++) {
						if(adj[i][j] == 1 ) adj[cur][j] =1;
					}
				}
			}
		}
		
		int cnt = 0;
		for (int j = 0; j <= N; j++) {
			cnt += adj[cur][j];
		}
		adj[cur][0] = cnt;
		
	}
	
	
}
```
