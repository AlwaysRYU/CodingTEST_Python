package hwalgo05_부울경_3반_류기탁;

import java.io.BufferedReader;
import java.io.FileInputStream;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class Solution {
	static int answer;
	static int[] array;
	static int[] src;
	static int max;
	static int M;
	

	
	public static void main(String[] args) throws Exception {
		System.setIn(new FileInputStream("./src/hwalgo05_부울경_3반_류기탁/input.txt"));//이렇게하자.
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		int Number = Integer.parseInt(br.readLine());
		int test = 0;
		while(test++ < Number) {
			answer = 0;
			StringTokenizer st = new StringTokenizer(br.readLine());
			int N = Integer.parseInt(st.nextToken());
			array = new int[N];
			M = Integer.parseInt(st.nextToken());
			StringTokenizer arr = new StringTokenizer(br.readLine());
			for (int i = 0; i < N; i++) {
				array[i] = Integer.parseInt(arr.nextToken());
			}
			//
			System.out.println( "최대 용량 : " + M );
			max = 0;
//			comb(0,0,0);
			for (int i = 0; i < N; i++) {
				for (int j = i+1; j < N; j++) {
					int temp = array[i]+array[j];
					if (temp > M )
						continue;
					max = Math.max(temp, max);
				}
			}
			
			if ( max == 0 ) {
				max = -1;
			}
			//출력 
			System.out.println(Arrays.toString(array));
			System.out.print("#" + test + " ");
			System.out.print(max + " ");
			
			System.out.println();
		}

	}
	
	
	static void comb(int srcI, int kg, int count ) {
	// 
	if ( kg > M ) {
		return;
	}
	// 기저 조건
	if (count == 2) {
		max = Math.max(kg, max);
		return;
	}	
		//srcidx를 선택한는 경우
		comb(srcI + 1, kg + array[srcI], count +1 );
		//그렇지 않은 경우
		comb(srcI + 1, kg + array[srcI], count);
	}
}
