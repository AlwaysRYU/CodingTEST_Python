import java.io.BufferedReader;
import java.io.FileInputStream;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class D24_오셀로 {
	static int BlackN, WhiteN, Test;
	static int N, M;
	static int[][] field;
	
	public static void main(String[] args) throws Exception {
		System.setIn(new FileInputStream("./src/_202108/24.txt"));
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		Test = Integer.parseInt(br.readLine());
		for (int i = 1; i <= Test; i++) {
			StringTokenizer st = new StringTokenizer(br.readLine());
			N = Integer.parseInt(st.nextToken());
			M = Integer.parseInt(st.nextToken());
			field = new int[N][N];
			field[(N/2)-1][N/2] = field[N/2][(N/2)-1] = 1;
			field[(N/2)-1][(N/2)-1] = field[N/2][N/2] = 2;
			
			for (int j = 0; j < N; j++) {
				for (int j2 = 0; j2 < N; j2++) {
					System.out.print(field[j][j2] + " ");
				}
				System.out.println();
			}
			System.out.println();
			
			for (int j = 0; j < M; j++) {
				st = new StringTokenizer(br.readLine());
				int X = Integer.parseInt(st.nextToken()) -1;
				int Y = Integer.parseInt(st.nextToken()) -1;
				int Color = Integer.parseInt(st.nextToken());
				
				Stone(X,Y,Color);
				// 흑 - 1
				// 백 - 2
				
			}
			for (int j = 0; j < N; j++) {
				for (int j2 = 0; j2 < N; j2++) {
					System.out.print(field[j][j2] + " ");
				}
				System.out.println();
			}
			System.out.println();
			// 초기화
			BlackN = WhiteN = 0;
			for (int j = 0; j < N; j++) {
				for (int j2 = 0; j2 < N; j2++) {
					if(field[j][j2] == 1 ) BlackN += 1;
					else if (field[j][j2] == 2) WhiteN += 1;
				}
			}
			System.out.println("#" + i + " " + BlackN + " " + WhiteN);
		}
		
	}
	
	static void Stone(int x, int y, int Color) {
		// 돌 놓고
		field[x][y] = Color;
		int reverse = 0;
		if (Color == 1 ) reverse = 2;
		else if (Color == 2) reverse = 1;
		// 체크
		int[] dx = { -1, -1, -1, 0, 1, 1,  1,  0 };
		int[] dy = { -1,  0,  1, 1, 1, 0, -1, -1 };
		for (int i = 0; i < 8; i++) {
			int nx = x + dx[i];
			int ny = y + dy[i];
			if (nx<0 || ny<0 || nx >= N || ny>= N) continue;
			// 다른 경우에만 바꾸기 실행 
			if (field[nx][ny] == reverse ) {
				// 바꾸기 실행 
				int lx = 0;
				int ly = 0;
				boolean change = false;
				while (true) {
					nx += dx[i];
					ny += dy[i];
					
					if (nx<0 || ny<0 || nx >= N || ny>= N) {
						break;
					}
					/// 같은거 면 설정
					if (field[nx][ny] == 0) break;
					if (field[nx][ny] == Color) {
						lx = nx;
						ly = ny;
						change = true;
						nx = x + dx[i];
						ny = y + dy[i];
						break;
					}
				}// 마지막찾았다.
				
				if (change) {
					while(true) {
						field[nx][ny] = Color;
						nx += dx[i];
						ny += dy[i];
						if (nx == lx && ny == ly) break;
					}
				}//바꿨다.
				
				
			}
			
		}
	}
