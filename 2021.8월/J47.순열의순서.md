# 순열의 순서

#### 최종
수학적으로 생각하는 것이 중요하다.
```java

package _202108;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.Scanner;
import java.util.StringTokenizer;

public class D18_순열의순서3 {
	// N 1~20
	static int N;
	static int Q;
	static int K;
	static long[] fac;
	static int[] answer;
	static boolean[] check;
	
	
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		N = Integer.parseInt(br.readLine());
		
		StringTokenizer st = new StringTokenizer(br.readLine());
		Q = Integer.parseInt(st.nextToken());

		check = new boolean[21];
		fac = new long[21];
		
		
		fac[0] = 1; // 
		for (int i = 1; i < 21; i++) {
			fac[i] = i * fac[i-1];
		}
		
		if (Q == 1) {
			// 1번 경우
			
			long find = Long.parseLong(st.nextToken());
			answer = new int[N]; // 답 
			
			for (int i = 0; i < N; i++) {
				for (int j = 1; j <= N; j++) {
					if (check[j] == true) continue; //쓴거면넘김
					if (find > fac[N-i-1]){
						find -= fac[N-i-1];
					} else {
						answer[i] = j;
						check[j] = true;
						break;
					}
				}
			}
			for (int j2 = 0; j2 < N; j2++) {
				System.out.print(answer[j2] + " ");
			}
			
		} else {
			//2번 경우
			answer = new int[N];
			for (int i = 0; i < N; i++) {
				answer[i] = Integer.parseInt(st.nextToken());
			}
			long result = 0;
			for (int i = 0; i < N; i++) {
				for (int j = 1; j < answer[i]; j++) {
					if (check[j] == true) continue;
					
					result += fac[N-1-i];
					
					
				}
				check[answer[i]] = true;
			}
			System.out.println( (result+1));
			
		}
		
		
    }

	
	

}
```

```java
package _202108;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.StringTokenizer;

public class D18_순열의순서 {
	// N 1~20
	static int N;
	static int Q;
	static int K;
	static int[] SY;
	
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		N = Integer.parseInt(br.readLine());
		StringTokenizer st = new StringTokenizer(br.readLine());
		Q = Integer.parseInt(st.nextToken());
		SY = new int[N];
		for (int i = 1; i <= N; i++) {
			SY[i-1] = i;
		}
		
		if (Q == 1) {
			// 1. K를 입력 받고, 
			K = Integer.parseInt(st.nextToken());
			// 목적 : K번째 순열을 나타내는  N개의 수를 출력한다.
			Quest1(K);
		} else  {
			// 2. 임의 순열을 나타내는 N개의 수를 입력받는다.
			String SYstr = "";
			for (int i = 0; i < N; i++) {
				SYstr += st.nextToken();
			}
			// 편의를 위해서 str로 바꿔줌  
//			System.out.println(SYstr);
			// 목적 : 입력받은 순열은 몇번째 순열인가?
			Quest2(SYstr);
			
		}
		
		
	}
	
	// 인덱스로 받음
	static void change(int[] arr, int A, int B) {
		int temp = arr[A];
		arr[A] = arr[B];
		arr[B] = temp;
		
	}
	
	static void Quest1(int KK) {
		// 목적 : K번째 순열을 나타내는  N개의 수를 출력한다.
		
		// 맨처음 순열 생성 
		
		// 현재 번째 순열
		int Number = 1;
		while(true) {
			if (Number == KK ) {
				// 정답 찾기 
				for(int x : SY) {
					System.out.print( x + " ");
				}
//				System.out.println(Arrays.toString(SY));
				break;
			}
			
			//아니면 진행 
			
			// 1. 맨끝 에서부터 시작
			int index = N-1;
			while( index > 0 
					&& SY[index-1] >= SY[index]) {
				// 왼쪽요소 > 오른쪽요소 인덱스 찾기(오른쪽이 기준)
				index -= 1;
			}
			if ( index == 0 ) {
				// 맨첫번째일경우는 아마 없다.
				System.out.println(Arrays.toString(SY));
			}
			
			// 2. 
			int index2 = N-1;
			while( SY[index-1] >= SY[index2]) index2 -= 1;
			
			change(SY, index-1, index2);
			
			// 3.  나머지를 역순으로
			// 즉, 서로 바꿔주면된다.
			int index3 = N-1;
			while( index < index3) {
				change(SY, index,index3);
				index += 1;
				index3 -= 1;
			}
			
			Number += 1;
		} ///  반복문 끝 
		
	}
	
	static void Quest2(String find) {

		int Number = 1;
		while(true) {

			temp = temp.substring( 1, temp.length()-1).replace(", " ,"");

				// 왼쪽요소 > 오른쪽요소 인덱스 찾기(오른쪽이 기준)
```
