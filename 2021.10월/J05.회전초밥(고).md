### 풀이
* 재밌는 문제이다. 더 간소화 한다면 할 수 있을 것 같다.
```java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.StringTokenizer;

public class Main {
	static int N; // 접시수
	static int d, k, c; // 초밥 가짓수 d 연속해서먹는접시수 k 쿠폰번호 c
	static int[] arr; // 전체 배열
	
	static int[] susi;
	static int answer = 0; // 초밥 가짓수
	
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine());
		N = Integer.parseInt(st.nextToken());
		d = Integer.parseInt(st.nextToken());
		k = Integer.parseInt(st.nextToken());
		c = Integer.parseInt(st.nextToken());
		
		// #. 초기화 및 입력
		// 배열은 연속먹는길이 -1 + 초밥접시개수 추가
		arr = new int[N + (k-1)];
		for (int i = 0; i < N; i++) {
			int temp = Integer.parseInt(br.readLine());
			arr[i] = temp;
		}
		int index = N;
		int arrL = arr.length;
		while( index < arrL) {
			arr[index] = arr[index - N];
			index += 1;
		}
//		System.out.println(Arrays.toString(arr));
		
		// 맨처음은 직접
		susi = new int[3001];
		int count = 0;
		for (int i = 0; i < k; i++) {
			int now = arr[i];
			if ( susi[now] == 0 ) {
				count +=1;
			}
			susi[now] += 1;
		}
		
		// 쿠폰은 항상 넣어준다.
		if (susi[c] == 0 ) count += 1;
		susi[c] += 1;
		
		// 최대 경우
//		for (int j = 0; j < susi.length; j++) {
//			if (susi[j] != 0 ) System.out.print(j + " : " + susi[j] + " // ");
//		}
//		System.out.println();
//		System.out.println(count);
		answer = Math.max(count, answer);
		
		
		// 연산시작
		for (int i = 1; i < N-1; i++) {
			// 이전거 빼주기
			int before = i-1;
			susi[arr[before]] -= 1;
			if (susi[arr[before]] == 0 ) count -= 1;
			
			// 다음거 넣어주기
			int next = i + k-1;
			if (susi[arr[next]] == 0) count += 1;
			susi[arr[next]] += 1;
			
			// 최대값 갱신
//			answer = Math.max(count, answer);
			if ( count > answer ) {
				answer = count;
//				for (int j = 0; j < susi.length; j++) {
//					if (susi[j] != 0 ) System.out.print(j + " : " + susi[j] + " // ");
//				}
//				System.out.println();
//				System.out.println("최대값 갱신 : " + answer);
			}
		}
		
		System.out.println(answer);
		
		
	}

}
```
